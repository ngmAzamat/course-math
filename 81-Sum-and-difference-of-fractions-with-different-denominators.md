$$
\frac{5}{3a} - \frac{2}{9a}
= \frac{5 × 3}{3 × 3a} - \frac{2}{9a}
= \frac{15}{9a} - \frac{2}{9a}
= \frac{15 - 2}{9a}
= \frac{13}{9a}
$$

\frac{2}{a} + \frac{12}{a(a - 6)} - \frac{1 - a}{a - 6}
= \frac{(2 × (a - 6)) + (12) - (1 - a × a)}{a × (a - 6)}
= \frac{2(a - 6) + 12 - a(1 - a)}{a(a - 6)}
= \frac{2a - 12 + 12 - a + a^2}{a(a - 6)}
= \frac{a + a^2}{a(a - 6)}
= \frac{a(1 + a)}{a(a - 6)}
= \frac{1 + a}{a - 6}
