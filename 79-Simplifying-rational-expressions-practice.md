$$
\frac{a^2 + 2ab + b^2 - c^2}{a^2 + 2ac + c^2 - b^2}
= \frac{(a^2 + 2ab + b^2) - c^2}{(a^2 + 2ac + c^2) - b^2}
= \frac{(a + b)^2 - c^2}{(a + c)^2 - b^2}
= \frac{(a + b - c)(a + b + c)}{(a + c - b)(a + c + b)}
= \frac{(a + b - c)(a + b + c)}{(a + c - b)(a + c + b)}
= \frac{a + b - c}{a + c - b}
$$
