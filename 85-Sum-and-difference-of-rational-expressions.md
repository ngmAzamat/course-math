= \frac{1}{2(3x - 2y)} - \frac{1}{2(3x + 2y)} - \frac{3x}{(2y - 3x)(2y + 3x)}
= \frac{1}{2(3x - 2y)} - \frac{1}{2(3x + 2y)} + \frac{3x}{(3x - 2y)(3x + 2y)}
= \frac{1}{2(3x - 2y)} - \frac{1}{2(3x + 2y)} + \frac{3x}{(3x - 2y)(3x + 2y)}
= \frac{(1 × (3x + 2y)) - (1 × (3x - 2y)) + (3x × 2)}{2(3x - 2y)(3x + 2y)}
= \frac{(3x + 2y) - (3x - 2y) + (3x × 2)}{2(3x - 2y)(3x + 2y)}
= \frac{3x + 2y - 3x + 2y + 3x × 2}{2(3x - 2y)(3x + 2y)}
= \frac{4y + 6x}{2(3x - 2y)(3x + 2y)}
