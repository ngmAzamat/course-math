$$ 4x(a - b) - a + b = 4x(a - b) - (a - b) = (4x - 1)(a - b) $$

$$ x(a - b) + a - b = (x)(a - b) + (a - b) = (x + 1)(a - b) $$

$$ 3a(x + y) - x - y = 3a(x + y) - (x + y) = (3a - 1)(x + y) $$

$$ 2a(x - y) - (y - x) = (2a + 1)(x - y) $$

$$ a^2 - b^2 - a + b = (a^2 - b^2) - (a - b) = (a - b)(a + b - 1) $$
